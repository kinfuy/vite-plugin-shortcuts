## Installation

```sh
npm i -D vite-plugin-shortcuts
```

## Usage

```ts
import { defineConfig } from 'vite';
import { shortcutsPlugin } from 'vite-plugin-tsc-watch';

export default defineConfig({
  plugins: [shortcutsPlugin()],
});
```
